
// ================================================================
// InputBar — 标签行 + 输入框 + @mention (提取自 MainScreen)
// ================================================================

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun InputBar(
    modifier: Modifier = Modifier,
    inputText: String,
    onInputChange: (String) -> Unit,
    inputEnabled: Boolean,
    inputFocus: androidx.compose.ui.focus.FocusRequester,
    activeTags: List<InputTag>,
    onRemoveTag: (InputTag) -> Unit,
    showMentionDropdown: Boolean,
    mentionQuery: String,
    onMentionStateChange: (query: String, show: Boolean) -> Unit,
    showExpandSheet: Boolean,
    onExpandSheetChange: (Boolean) -> Unit,
    settingsViewModel: SettingsViewModel?,
    viewModel: AgentViewModel,
    pluginViewModel: PluginViewModel,
    strings: com.mengpaw.shell.ui.localization.AppStrings
) {
    val isTablet = isWide()
    val padBottom = if (isTablet) 12.dp else 0.dp
    val barWidth = if (isTablet) 0.8f else 1f
    val shape = if (isTablet) RoundedCornerShape(ArcoRadius.xl) else RoundedCornerShape(0.dp)

    Column(modifier.fillMaxWidth(barWidth)) {
        if (activeTags.isNotEmpty()) {
            Row(Modifier.fillMaxWidth().padding(horizontal = ArcoSpacing.lg, vertical = ArcoSpacing.xs), horizontalArrangement = Arrangement.spacedBy(ArcoSpacing.xs)) {
                activeTags.forEach { tag ->
                    val chipLabel = when (tag) { is InputTag.Mode -> tag.mode.prefix; is InputTag.AgentRef -> "@${tag.agentName}" }
                    AssistChip(onClick = {}, label = { Text(chipLabel, style = MaterialTheme.typography.labelSmall) },
                        trailingIcon = { Icon(Icons.Filled.Close, strings.tagDismiss, Modifier.size(14.dp).clickable { onRemoveTag(tag) }) },
                        shape = RoundedCornerShape(ArcoRadius.sm),
                        colors = AssistChipDefaults.assistChipColors(containerColor = ThemeColors.brandContainer, labelColor = ThemeColors.brand))
                }
            }
        }
        Box(Modifier.padding(bottom = padBottom)) {
            Surface(shadowElevation = 8.dp, color = ThemeColors.bgPrimary, shape = shape) {
                Row(Modifier.fillMaxWidth().imePadding().navigationBarsPadding().padding(horizontal = 12.dp, vertical = ArcoSpacing.sm), verticalAlignment = Alignment.CenterVertically) {
                    Box(Modifier.size(44.dp).background(ThemeColors.surfaceContainerHigh, CircleShape).clickable { onExpandSheetChange(true) }, contentAlignment = Alignment.Center) {
                        Icon(Icons.Outlined.Add, "扩展", tint = ThemeColors.textSecondary, modifier = Modifier.size(24.dp))
                    }
                    val keyMaxSteps = settingsViewModel?.state?.value?.maxSteps ?: 50
                    fun doSend() {
                        if (inputText.isNotBlank()) {
                            val t = inputText; onInputChange("")
                            val mt = activeTags.filterIsInstance<InputTag.Mode>().firstOrNull()
                            val at = activeTags.filterIsInstance<InputTag.AgentRef>().firstOrNull()
                            viewModel.submitTask(t, pluginViewModel, maxSteps = keyMaxSteps, executionMode = mt?.mode, agentRef = at?.agentName)
                            inputFocus.requestFocus()
                        }
                    }
                    OutlinedTextField(value = inputText, onValueChange = { v ->
                        onInputChange(v)
                        val ai = v.lastIndexOf('@')
                        if (ai >= 0) { val ba = if (ai > 0) v[ai - 1] else ' '; if (ba == ' ' || ba == '\n' || ai == 0) { val q = v.substring(ai + 1); if (!q.contains(' ') && !q.contains('\n')) onMentionStateChange(q, true) else onMentionStateChange("", false) } else onMentionStateChange("", false) } else onMentionStateChange("", false)
                    }, modifier = Modifier.weight(1f).padding(horizontal = 4.dp).focusRequester(inputFocus).onPreviewKeyEvent { e ->
                        if (e.nativeKeyEvent.keyCode == android.view.KeyEvent.KEYCODE_ENTER) { if (e.nativeKeyEvent.isCtrlPressed || e.nativeKeyEvent.isShiftPressed) onInputChange(inputText + "\n") else doSend(); true } else false
                    }, enabled = inputEnabled,
                        placeholder = { Text(when (activeTags.filterIsInstance<InputTag.Mode>().firstOrNull()?.mode) { ExecutionMode.MISSION -> strings.placeholderMission; ExecutionMode.RESEARCH -> strings.placeholderResearch; ExecutionMode.TRANSLATE -> strings.placeholderTranslate; ExecutionMode.DREAM -> strings.placeholderDream; else -> strings.inputPlaceholder }) },
                        shape = RoundedCornerShape(ArcoRadius.lg), colors = OutlinedTextFieldDefaults.colors(focusedBorderColor = ThemeColors.brand, unfocusedBorderColor = ThemeColors.border), minLines = 1, maxLines = 4,
                        keyboardOptions = KeyboardOptions.Default.copy(imeAction = ImeAction.Default), keyboardActions = KeyboardActions(onSend = { doSend() }))
                    val scope = rememberCoroutineScope(); val ay = remember { Animatable(0f) }; val aa = remember { Animatable(1f) }
                    Box(Modifier.size(44.dp).background(if (inputEnabled) ThemeColors.brand else ThemeColors.brand.copy(alpha = 0.38f), CircleShape).clickable(enabled = inputEnabled) {
                        val t = inputText; if (t.isNotBlank()) { onInputChange(""); inputFocus.requestFocus(); val mt = activeTags.filterIsInstance<InputTag.Mode>().firstOrNull(); val at = activeTags.filterIsInstance<InputTag.AgentRef>().firstOrNull()
                            scope.launch { launch { ay.animateTo(-60f, tween(280)) }; launch { aa.animateTo(0f, tween(280)) }; ay.snapTo(60f); viewModel.submitTask(t, pluginViewModel, maxSteps = keyMaxSteps, executionMode = mt?.mode, agentRef = at?.agentName); launch { ay.animateTo(0f, tween(280)) }; launch { aa.animateTo(1f, tween(280)) } } }
                    }, contentAlignment = Alignment.Center) { Icon(if (inputText.isEmpty()) Icons.Outlined.ArrowUpward else Icons.Filled.ArrowUpward, "发送", tint = Color.White, modifier = Modifier.offset(y = ay.value.dp).alpha(aa.value)) }
                }
            }
        }
        val mentionAgents = remember(mentionQuery, viewModel.agentNamesForMention().size) { viewModel.agentNamesForMention().filter { (n, _) -> mentionQuery.isBlank() || n.contains(mentionQuery, ignoreCase = true) } }
        if (showMentionDropdown && mentionAgents.isNotEmpty()) {
            DropdownMenu(expanded = true, onDismissRequest = { onMentionStateChange("", false) }) {
                mentionAgents.take(6).forEach { (name, fw) ->
                    DropdownMenuItem(text = { Row(verticalAlignment = Alignment.CenterVertically) { Text("@$name", fontWeight = FontWeight.Bold, style = MaterialTheme.typography.bodyMedium); if (fw != null) { Spacer(Modifier.width(4.dp)); Text("· $fw", style = MaterialTheme.typography.labelSmall, color = ThemeColors.textSecondary) } } },
                        onClick = { val c = inputText; val ai = c.lastIndexOf('@'); if (ai >= 0) { onInputChange(c.substring(0, ai) + "@$name " + c.substring(ai + 1 + mentionQuery.length)) }; onMentionStateChange("", false); viewModel.addTag(InputTag.AgentRef(name)); try { inputFocus.requestFocus() } catch (_: Exception) {} },
                        leadingIcon = { Surface(shape = CircleShape, color = ThemeColors.brandContainer, modifier = Modifier.size(24.dp)) { Box(contentAlignment = Alignment.Center) { Text(name.take(1), color = ThemeColors.brand, fontWeight = FontWeight.Bold, fontSize = 12.sp) } } })
                }
            }
        }
    }
}
